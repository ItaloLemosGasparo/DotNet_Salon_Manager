CREATE TABLE Passwords (
    PasswordId INT IDENTITY PRIMARY KEY,
    UserId INT FOREIGN KEY REFERENCES Users(UserId) NOT NULL,
    PasswordHash VARCHAR(64) NOT NULL,
    PasswordSalt VARCHAR(64) NOT NULL
);
