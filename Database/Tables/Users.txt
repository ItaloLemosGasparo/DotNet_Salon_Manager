CREATE TABLE Users (
    UserId INT IDENTITY PRIMARY KEY,
    Name VARCHAR(100) NOT NULL,
    CPF VARCHAR(11) NOT NULL,
    Email VARCHAR(255) NOT NULL,
    AccessLevelId INT FOREIGN KEY REFERENCES AccessLevels(AccessLevelId) NOT NULL
);
